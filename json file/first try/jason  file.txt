jason : Javascript object notation 
--> Data repesentaion Format
-->commonly used for API And Config file
-->Lightweight and easay to read and write   
***Jason type 
    string ="hello zain "
    number = 10, 1.5 , -30
    Boolean =true , false 
    null    =   null 
    Array   = [1, 2, 3 ], ["ueyw", "636"]
    Object  = {"key" : "value"}

create file with .json extension 

  user.json
